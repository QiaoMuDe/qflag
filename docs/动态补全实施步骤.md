```go
// completeCmdName 动态补全子命令名
const completeCmdName = "__complete"

// createCompleteCmd 创建动态补全子命令
//
// 返回值:
//   - *Cmd: 动态补全子命令实例
//
// 功能说明:
//   - 创建一个隐藏的子命令，用于执行动态补全
//   - 接收当前命令行作为参数
//   - 计算并返回补全结果
func createCompleteCmd() *Cmd {
    cmd := NewCmd(completeCmdName, "", ExitOnError)
    cmd.SetDesc("内部命令：执行动态补全")
    cmd.SetHidden(true) // 隐藏子命令，不在帮助中显示
    
    // 设置执行函数
    cmd.SetRun(func(c types.Command) error {
        args := c.Args()
        if len(args) < 2 {
            return fmt.Errorf("用法: __complete <wordToComplete> <previousWord> [args...]")
        }
        
        wordToComplete := args[0]
        previousWord := args[1]
        cmdArgs := args[2:]
        
        // 获取根命令（通过父命令链找到根）
        rootCmd := findRootCmd(c)
        
        // 构建补全上下文
        ctx := completion.BuildContext(rootCmd, wordToComplete, previousWord, cmdArgs)
        
        // 执行补全
        result := completion.Execute(ctx)
        
        // 输出结果
        completion.PrintResult(result)
        
        return nil
    })
    
    return cmd
}
```

```go
// findRootCmd 找到根命令
func findRootCmd(cmd types.Command) types.Command {
    for {
        if cmd.IsRootCmd() {
            return cmd
        }
        // 通过父命令引用找到根
        if c, ok := cmd.(*Cmd); ok && c.parent != nil {
            cmd = c.parent
        } else {
            return cmd
        }
    }
}
```

## Phase 3: 补全核心逻辑

### 3.1 internal/completion/context.go

**操作**: 创建补全上下文文件

```go
package completion

import (
    "strings"
    
    "gitee.com/MM-Q/qflag/internal/types"
)

// Context 补全上下文
type Context struct {
    RootCmd        types.Command
    CurrentCmd     types.Command
    WordToComplete string
    PreviousWord   string
    Args           []string
}

// BuildContext 构建补全上下文
//
// 参数:
//   - rootCmd: 根命令
//   - wordToComplete: 当前正在补全的词
//   - previousWord: 前一个词
//   - args: 完整命令行参数
//
// 返回值:
//   - *Context: 补全上下文
func BuildContext(rootCmd types.Command, wordToComplete, previousWord string, args []string) *Context {
    ctx := &Context{
        RootCmd:        rootCmd,
        WordToComplete: wordToComplete,
        PreviousWord:   previousWord,
        Args:           args,
    }
    
    // 解析当前命令
    ctx.CurrentCmd = resolveCurrentCmd(rootCmd, args)
    
    return ctx
}

// resolveCurrentCmd 解析当前命令
//
// 参数:
//   - rootCmd: 根命令
//   - args: 命令行参数
//
// 返回值:
//   - types.Command: 当前命令
func resolveCurrentCmd(rootCmd types.Command, args []string) types.Command {
    current := rootCmd
    for _, arg := range args {
        // 遇到标志停止解析
        if strings.HasPrefix(arg, "-") {
            break
        }
        // 查找子命令
        if subCmd, ok := current.GetSubCmd(arg); ok {
            current = subCmd
        } else {
            break
        }
    }
    return current
}
```

### 3.2 internal/completion/execute.go

**操作**: 创建补全执行逻辑文件

```go
package completion

import (
    "fmt"
    "os"
    "strings"
    
    "gitee.com/MM-Q/qflag/internal/types"
)

// Execute 执行补全
//
// 参数:
//   - ctx: 补全上下文
//
// 返回值:
//   - types.CompletionResult: 补全结果
func Execute(ctx *Context) types.CompletionResult {
    // 检查是否有自定义补全函数
    if ctx.PreviousWord != "" && strings.HasPrefix(ctx.PreviousWord, "-") {
        flagName := strings.TrimLeft(ctx.PreviousWord, "-")
        if completer, ok := ctx.CurrentCmd.GetCompleter(flagName); ok {
            return completer(buildCompleterContext(ctx))
        }
    }
    
    // 默认补全逻辑
    return defaultComplete(ctx)
}

// buildCompleterContext 构建补全函数上下文
func buildCompleterContext(ctx *Context) types.CompletionContext {
    return types.CompletionContext{
        WordToComplete: ctx.WordToComplete,
        PreviousWord:   ctx.PreviousWord,
        Args:           ctx.Args,
    }
}

// defaultComplete 默认补全
//
// 补全优先级:
//   1. 标志值补全（如果前一个词是标志）
//   2. 标志名补全（如果当前词以 - 开头）
//   3. 子命令补全（如果当前词匹配子命令前缀）
//   4. 路径补全（默认兜底）
func defaultComplete(ctx *Context) types.CompletionResult {
    var completions []types.Completion
    
    // 判断是否正在补全标志值
    if ctx.PreviousWord != "" && strings.HasPrefix(ctx.PreviousWord, "-") {
        completions = completeFlagValue(ctx)
    } else if strings.HasPrefix(ctx.WordToComplete, "-") {
        // 补全标志名（长标志和短标志）
        completions = completeFlagName(ctx)
    } else {
        // 先尝试补全子命令
        completions = completeSubCommand(ctx)
        
        // 如果没有子命令匹配，补全路径
        if len(completions) == 0 {
            completions = completePath(ctx.WordToComplete)
        }
    }
    
    return types.CompletionResult{
        Completions: completions,
        Directive:   0,
    }
}

// completeFlagName 补全标志名（长标志和短标志）
func completeFlagName(ctx *Context) []types.Completion {
    var completions []types.Completion
    word := ctx.WordToComplete
    
    for _, flag := range ctx.CurrentCmd.Flags() {
        // 补全长标志
        if longName := flag.LongName(); longName != "" {
            if strings.HasPrefix(longName, word) {
                completions = append(completions, types.Completion{
                    Value:       longName,
                    Description: flag.Desc(),
                })
            }
        }
        
        // 补全短标志
        if shortName := flag.ShortName(); shortName != "" {
            shortFlag := "-" + shortName
            if strings.HasPrefix(shortFlag, word) {
                completions = append(completions, types.Completion{
                    Value:       shortFlag,
                    Description: flag.Desc(),
                })
            }
        }
    }
    return completions
}

// completeSubCommand 补全子命令
func completeSubCommand(ctx *Context) []types.Completion {
    var completions []types.Completion
    for _, subCmd := range ctx.CurrentCmd.SubCmds() {
        // 跳过隐藏子命令（如 __complete）
        if subCmd.IsHidden() {
            continue
        }
        name := subCmd.Name()
        if strings.HasPrefix(name, ctx.WordToComplete) {
            completions = append(completions, types.Completion{
                Value:       name,
                Description: subCmd.Desc(),
            })
        }
    }
    return completions
}

// completeFlagValue 补全标志值
func completeFlagValue(ctx *Context) []types.Completion {
    var completions []types.Completion
    
    flagName := strings.TrimLeft(ctx.PreviousWord, "-")
    if flag, ok := ctx.CurrentCmd.GetFlag(flagName); ok {
        // 如果是枚举类型，返回枚举值
        if enumValues := flag.EnumValues(); len(enumValues) > 0 {
            for _, opt := range enumValues {
                if strings.HasPrefix(opt, ctx.WordToComplete) {
                    completions = append(completions, types.Completion{Value: opt})
                }
            }
            return completions
        }
    }
    
    // 默认补全路径
    return completePath(ctx.WordToComplete)
}

// completePath 补全文件路径
func completePath(word string) []types.Completion {
    var completions []types.Completion
    
    // 解析路径前缀
    dir := "."
    prefix := ""
    if idx := strings.LastIndex(word, "/"); idx >= 0 {
        dir = word[:idx]
        if dir == "" {
            dir = "/"
        }
        prefix = word[idx+1:]
    } else if idx := strings.LastIndex(word, "\\"); idx >= 0 {
        dir = word[:idx]
        if dir == "" {
            dir = "\\"
        }
        prefix = word[idx+1:]
    } else {
        prefix = word
    }
    
    // 读取目录内容
    entries, err := os.ReadDir(dir)
    if err != nil {
        return completions
    }
    
    for _, entry := range entries {
        name := entry.Name()
        if strings.HasPrefix(name, prefix) {
            value := name
            if entry.IsDir() {
                value += "/"
            }
            completions = append(completions, types.Completion{Value: value})
        }
    }
    
    return completions
}

// PrintResult 打印补全结果
func PrintResult(result types.CompletionResult) {
    for _, comp := range result.Completions {
        if comp.Description != "" {
            fmt.Printf("%s\t%s\n", comp.Value, comp.Description)
        } else {
            fmt.Println(comp.Value)
        }
    }
    fmt.Printf(":%d\n", result.Directive)
}
```

### 3.3 internal/completion/completion.go

**操作**: 修改 Generate 函数，支持动态脚本生成

```go
// Generate 生成补全脚本
//
// 参数:
//   - cmd: 命令实例
//   - shellType: Shell 类型
//
// 返回值:
//   - string: 生成的脚本
//   - error: 错误
func Generate(cmd types.Command, shellType string) (string, error) {
    config := cmd.Config()
    
    // 检查是否启用补全
    if !config.Completion {
        return "", fmt.Errorf("completion not enabled")
    }
    
    // 【修改】根据配置决定生成静态还是动态脚本
    if config.DynamicCompletion {
        return generateDynamic(cmd, shellType)
    }
    
    return generateStatic(cmd, shellType)
}

// generateDynamic 生成动态补全脚本
//
// 参数:
//   - cmd: 命令实例
//   - shellType: Shell 类型
//
// 返回值:
//   - string: 生成的脚本
//   - error: 错误
func generateDynamic(cmd types.Command, shellType string) (string, error) {
    // 根据 shell 类型选择模板
    var tmplName string
    switch shellType {
    case "bash":
        tmplName = "bash_dynamic.tmpl"
    case "zsh":
        tmplName = "zsh_dynamic.tmpl"
    case "pwsh", "powershell":
        tmplName = "pwsh_dynamic.tmpl"
    default:
        return "", fmt.Errorf("unsupported shell type: %s", shellType)
    }
    
    // 加载模板
    tmplContent, err := templates.FS.ReadFile(tmplName)
    if err != nil {
        return "", fmt.Errorf("failed to load template: %w", err)
    }
    
    // 解析模板
    tmpl, err := template.New(tmplName).Parse(string(tmplContent))
    if err != nil {
        return "", fmt.Errorf("failed to parse template: %w", err)
    }
    
    // 准备数据
    data := struct {
        ProgramName string
    }{
        ProgramName: cmd.Name(),
    }
    
    // 执行模板
    var buf strings.Builder
    if err := tmpl.Execute(&buf, data); err != nil {
        return "", fmt.Errorf("failed to execute template: %w", err)
    }
    
    return buf.String(), nil
}
```

### 4.1 internal/completion/templates/bash_dynamic.tmpl

**操作**: 创建 Bash 动态补全模板

```bash
# {{.ProgramName}} completion script (dynamic)
# Generated by qflag

_{{.ProgramName}}_completion() {
    local cur prev words cword
    _init_completion || return

    # 调用 __complete 子命令获取补全结果
    # 参数: <当前词> <前一个词> <完整命令行>
    local completions
    completions=$({{.ProgramName}} __complete "$cur" "$prev" ${words[@]} 2>/dev/null)
    
    # 解析结果（最后一行是 :指令）
    local directive
    directive=$(echo "$completions" | tail -n 1)
    
    # 提取候选值（去掉最后一行指令）
    local candidates
    candidates=$(echo "$completions" | head -n -1)
    
    # 设置补全结果
    COMPREPLY=($(compgen -W "$candidates" -- "$cur"))
    
    return 0
}

# 注册补全函数
complete -F _{{.ProgramName}}_completion {{.ProgramName}}
```

### 4.2 internal/completion/templates/pwsh_dynamic.tmpl

**操作**: 创建 PowerShell 动态补全模板

```powershell
# {{.ProgramName}} completion script (dynamic)
# Generated by qflag

Register-ArgumentCompleter -Native -CommandName {{.ProgramName}} -ScriptBlock {
    param($wordToComplete, $commandAst, $cursorPosition)
    
    # 获取完整的命令行参数
    $words = $commandAst.CommandElements | ForEach-Object { $_.ToString() }
    
    # 获取前一个词
    $prev = ""
    if ($words.Count -gt 1) {
        $prev = $words[$words.Count - 2]
    }
    
    # 调用 __complete 子命令获取补全结果
    # 参数: <当前词> <前一个词> <完整命令行>
    $completions = & {{.ProgramName}} __complete $wordToComplete $prev @words 2>$null
    
    if (-not $completions) {
        return
    }
    
    # 解析结果（最后一行是 :指令）
    $lines = $completions -split "`n"
    $directive = $lines[-1]
    $candidates = $lines[0..($lines.Count - 2)]
    
    # 返回候选值
    foreach ($candidate in $candidates) {
        if ($candidate -match "^$wordToComplete") {
            # 处理带描述的格式: value`tdescription
            $parts = $candidate -split "`t", 2
            [System.Management.Automation.CompletionResult]::new(
                $parts[0],
                $parts[0],
                'ParameterValue',
                if ($parts[1]) { $parts[1] } else { $parts[0] }
            )
        }
    }
}
```

## 常见问题

### Q: 动态补全和静态补全有什么区别？

**A**: 
- **静态补全**: 预生成包含所有子命令和标志的脚本，补全时不调用程序
- **动态补全**: 脚本调用程序的 `__complete` 子命令实时获取补全结果

### Q: 为什么需要 `__complete` 子命令？

**A**: 动态补全需要访问完整的命令树和解析后的参数，通过子命令可以确保解析流程完整执行。

### Q: 自定义补全函数什么时候被调用？

**A**: 当补全某个标志的值时，如果该标志注册了自定义补全函数，会优先调用该函数。
