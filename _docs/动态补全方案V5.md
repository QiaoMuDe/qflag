# qflag 动态补全方案 V5

## 设计思路

参考现有静态补全脚本的完整功能，将核心补全逻辑（命令树遍历、模糊匹配、枚举值补全等）全部移到 Go 端实现。Shell 脚本只负责：
1. 接收用户输入
2. 调用程序获取补全结果
3. 展示补全列表

## 核心架构

```
用户按 Tab
    ↓
Shell 脚本收集上下文（当前词、前一个词、已输入的命令行）
    ↓
调用: myapp __complete <wordToComplete> <previousWord> <args...>
    ↓
Go 程序解析命令树，执行补全逻辑
    ↓
输出补全结果（每行一个候选，最后以 :指令 结尾）
    ↓
Shell 解析结果，展示给用户
```

## 补全逻辑（Go 端实现）

### 1. 补全优先级（与静态脚本一致）

```
第1步: 判断当前输入类型
    ├── 正在输入标志值（前一个词是标志，当前词不以 - 开头）
    │   ├── 枚举类型 → 补全枚举值
    │   └── 其他类型 → 补全路径
    │
    └── 正在输入命令/标志（当前词以 - 开头或为空）
        ├── 精确前缀匹配
        ├── 大小写不敏感前缀匹配
        ├── 子字符串匹配
        └── 模糊匹配（按分数排序）
```

| 模块 | 功能 | 对应静态脚本 |

| `CommandTree` | 构建命令树，支持子命令查找 | `cmd_tree` 数组 |

## Shell 脚本（极简版）

### Bash 动态模板

```bash
#!/usr/bin/env bash

_{{.ProgramName}}_complete() {
    local cur prev words cword
    _init_completion || return
    
    # 调用程序获取补全结果
    local completions
    completions=$({{.ProgramName}} __complete "$cur" "$prev" ${words[@]} 2>/dev/null)
    
    # 解析结果（最后一行是 :指令）
    local directive
    directive=$(echo "$completions" | tail -n 1)
    
    # 提取候选值（去掉最后一行指令）
    local candidates
    candidates=$(echo "$completions" | head -n -1)
    
    # 设置补全结果
    COMPREPLY=($(compgen -W "$candidates" -- "$cur"))
}

complete -F _{{.ProgramName}}_complete {{.ProgramName}}
```

### PowerShell 动态模板

```powershell
# {{.ProgramName}} completion script (dynamic)

Register-ArgumentCompleter -Native -CommandName {{.ProgramName}} -ScriptBlock {
    param($wordToComplete, $commandAst, $cursorPosition)
    
    # 获取命令行参数
    $words = $commandAst.CommandElements | ForEach-Object { $_.ToString() }
    $prev = if ($words.Count -gt 1) { $words[$words.Count - 2] } else { "" }
    
    # 调用程序获取补全结果
    $completions = & {{.ProgramName}} __complete $wordToComplete $prev @words 2>$null
    
    if (-not $completions) { return }
    
    # 解析结果
    $lines = $completions -split "`n"
    $directive = $lines[-1]
    $candidates = $lines[0..($lines.Count - 2)]
    
    # 返回候选值
    foreach ($candidate in $candidates) {
        if ($candidate -match "^$wordToComplete") {
            $parts = $candidate -split "`t", 2
            [System.Management.Automation.CompletionResult]::new(
                $parts[0], $parts[0], 'ParameterValue', 
                if ($parts[1]) { $parts[1] } else { $parts[0] }
            )
        }
    }
}
```

```go
    })
```

```go
// internal/completion/execute.go

package completion

import (
    "fmt"
    "os"
    "strings"
    
    "gitee.com/MM-Q/qflag/internal/types"
)

// RunComplete 执行动态补全（供 cmd 包调用）
func RunComplete(cmd types.Command, root types.Command) error {
    args := cmd.Args()
    if len(args) < 2 {
        return fmt.Errorf("用法: __complete <wordToComplete> <previousWord> [args...]")
    }
    
    wordToComplete := args[0]
    previousWord := args[1]
    cmdArgs := args[2:]
    
    // 构建补全上下文
    ctx := NewContext(root, wordToComplete, previousWord, cmdArgs)
    
    // 执行补全
    result := Execute(ctx)
    
    // 输出结果
    for _, comp := range result.Completions {
        if comp.Description != "" {
            fmt.Printf("%s\t%s\n", comp.Value, comp.Description)
        } else {
            fmt.Println(comp.Value)
        }
    }
    fmt.Printf(":%d\n", result.Directive)
    
    return nil
}
```

```go
// internal/completion/context.go

package completion

import (
    "strings"
    
    "gitee.com/MM-Q/qflag/internal/types"
)

// Context 补全上下文
type Context struct {
    RootCmd        types.Command
    CurrentCmd     types.Command
    WordToComplete string
    PreviousWord   string
    Args           []string
}

// NewContext 创建补全上下文
func NewContext(root types.Command, wordToComplete, previousWord string, args []string) *Context {
    ctx := &Context{
        RootCmd:        root,
        WordToComplete: wordToComplete,
        PreviousWord:   previousWord,
        Args:           args,
    }
    
    // 解析当前命令
    ctx.CurrentCmd = resolveCurrentCmd(root, args)
    
    return ctx
}

// resolveCurrentCmd 解析当前命令
func resolveCurrentCmd(root types.Command, args []string) types.Command {
    current := root
    for _, arg := range args {
        if strings.HasPrefix(arg, "-") {
            break
        }
        if subCmd, ok := current.GetSubCmd(arg); ok {
            current = subCmd
        } else {
            break
        }
    }
    return current
}
```

```go
// internal/completion/dynamic.go

package completion

import (
    "os"
    "strings"
    
    "gitee.com/MM-Q/qflag/internal/types"
)

// Execute 执行补全
func Execute(ctx *Context) types.CompletionResult {
    // 判断当前输入类型
    if isTypingFlagValue(ctx) {
        return completeFlagValue(ctx)
    }
    
    return completeCommandOrFlag(ctx)
}

// isTypingFlagValue 判断是否正在输入标志值
func isTypingFlagValue(ctx *Context) bool {
    if ctx.PreviousWord == "" {
        return false
    }
    if !strings.HasPrefix(ctx.PreviousWord, "-") {
        return false
    }
    if strings.HasPrefix(ctx.WordToComplete, "-") {
        return false
    }
    return true
}

// completeFlagValue 补全标志值
func completeFlagValue(ctx *Context) types.CompletionResult {
    flagName := strings.TrimLeft(ctx.PreviousWord, "-")
    flag, ok := ctx.CurrentCmd.GetFlag(flagName)
    if !ok {
        return completePath(ctx.WordToComplete)
    }
    
    // 枚举类型
    if enumValues := flag.EnumValues(); len(enumValues) > 0 {
        return completeFromOptions(ctx.WordToComplete, enumValues)
    }
    
    // 其他类型默认补全路径
    return completePath(ctx.WordToComplete)
}

// completeCommandOrFlag 补全命令或标志
func completeCommandOrFlag(ctx *Context) types.CompletionResult {
    var options []string
    
    // 收集子命令
    for _, subCmd := range ctx.CurrentCmd.SubCmds() {
        if !subCmd.IsHidden() {
            options = append(options, subCmd.Name())
        }
    }
    
    // 收集标志
    for _, flag := range ctx.CurrentCmd.Flags() {
        if longName := flag.LongName(); longName != "" {
            options = append(options, "--"+longName)
        }
        if shortName := flag.ShortName(); shortName != "" {
            options = append(options, "-"+shortName)
        }
    }
    
    return completeFromOptions(ctx.WordToComplete, options)
}

// completeFromOptions 从选项列表中补全
func completeFromOptions(word string, options []string) types.CompletionResult {
    matches := IntelligentMatches(word, options)
    
    var completions []types.Completion
    for _, match := range matches {
        completions = append(completions, types.Completion{Value: match})
    }
    
    return types.CompletionResult{
        Completions: completions,
        Directive:   0,
    }
}

// completePath 补全文件路径
func completePath(word string) types.CompletionResult {
    var completions []types.Completion
    
    // 解析路径
    dir := "."
    prefix := word
    if idx := strings.LastIndexAny(word, "/\\"); idx >= 0 {
        dir = word[:idx]
        if dir == "" {
            dir = "/"
        }
        prefix = word[idx+1:]
    }
    
    // 读取目录
    entries, err := os.ReadDir(dir)
    if err != nil {
        return types.CompletionResult{Completions: completions}
    }
    
    for _, entry := range entries {
        name := entry.Name()
        if strings.HasPrefix(name, prefix) {
            value := name
            if entry.IsDir() {
                value += "/"
            }
            completions = append(completions, types.Completion{Value: value})
        }
    }
    
    return types.CompletionResult{Completions: completions}
}
```

### 5. 模糊匹配算法（Go 实现）

```go
// internal/completion/fuzzy.go

package completion

import (
    "math"
    "strings"
    "unicode"
)

// FuzzyScore 计算模糊匹配分数 (0-100)
func FuzzyScore(pattern, candidate string) int {
    patternLen := len(pattern)
    candidateLen := len(candidate)
    
    // 快速路径1: 空模式
    if patternLen == 0 {
        return 100
    }
    
    // 快速路径2: 长度检查
    if candidateLen < patternLen {
        return 0
    }
    
    // 快速路径3: 完全前缀匹配
    if strings.HasPrefix(strings.ToLower(candidate), strings.ToLower(pattern)) {
        return 100
    }
    
    // 转小写
    patternLower := strings.ToLower(pattern)
    candidateLower := strings.ToLower(candidate)
    
    // 字符存在性预检查
    for _, char := range patternLower {
        if !strings.ContainsRune(candidateLower, char) {
            return 0
        }
    }
    
    // 核心匹配算法
    matched := 0
    consecutive := 0
    maxConsecutive := 0
    candidatePos := 0
    startBonus := 0
    
    // 起始匹配奖励
    if strings.HasPrefix(candidateLower, patternLower) {
        startBonus = 20
    }
    
    // 逐字符匹配
    for i := 0; i < patternLen; i++ {
        patternChar := patternLower[i]
        found := false
        
        for j := candidatePos; j < candidateLen; j++ {
            if candidateLower[j] == patternChar {
                matched++
                found = true
                
                if j == candidatePos {
                    consecutive++
                    if consecutive > maxConsecutive {
                        maxConsecutive = consecutive
                    }
                } else {
                    consecutive = 1
                }
                
                candidatePos = j + 1
                break
            }
        }
        
        if !found {
            consecutive = 0
        }
    }
    
    // 评分计算
    baseScore := (matched * 60) / patternLen
    consecutiveBonus := (maxConsecutive * 20) / patternLen
    lengthPenalty := int(math.Min(float64(candidateLen-patternLen), 10))
    
    finalScore := baseScore + consecutiveBonus + startBonus - lengthPenalty
    
    if finalScore < 0 {
        return 0
    }
    if finalScore > 100 {
        return 100
    }
    return finalScore
}

// IntelligentMatches 智能匹配（渐进式策略）
func IntelligentMatches(pattern string, options []string) []string {
    if pattern == "" {
        return options
    }
    
    patternLower := strings.ToLower(pattern)
    
    // 第1级: 精确前缀匹配
    var exactMatches []string
    for _, opt := range options {
        if strings.HasPrefix(opt, pattern) {
            exactMatches = append(exactMatches, opt)
        }
    }
    if len(exactMatches) > 0 {
        return exactMatches
    }
    
    // 第2级: 大小写不敏感前缀匹配
    var caseInsensitiveMatches []string
    for _, opt := range options {
        if strings.HasPrefix(strings.ToLower(opt), patternLower) {
            caseInsensitiveMatches = append(caseInsensitiveMatches, opt)
        }
    }
    if len(caseInsensitiveMatches) > 0 {
        return caseInsensitiveMatches
    }
    
    // 第3级: 子字符串匹配
    var substringMatches []string
    for _, opt := range options {
        if strings.Contains(strings.ToLower(opt), patternLower) {
            substringMatches = append(substringMatches, opt)
        }
    }
    if len(substringMatches) > 0 {
        return substringMatches
    }
    
    // 第4级: 模糊匹配
    type scoredMatch struct {
        option string
        score  int
    }
    
    var scoredMatches []scoredMatch
    for _, opt := range options {
        score := FuzzyScore(pattern, opt)
        if score >= 25 { // 阈值
            scoredMatches = append(scoredMatches, scoredMatch{option: opt, score: score})
        }
    }
    
    // 按分数排序
    for i := 0; i < len(scoredMatches); i++ {
        for j := i + 1; j < len(scoredMatches); j++ {
            if scoredMatches[j].score > scoredMatches[i].score {
                scoredMatches[i], scoredMatches[j] = scoredMatches[j], scoredMatches[i]
            }
        }
    }
    
    // 取前10个
    var result []string
    for i := 0; i < len(scoredMatches) && i < 10; i++ {
        result = append(result, scoredMatches[i].option)
    }
    
    return result
}
```

```go
func GenerateScript(cmd types.Command, shell string, config *types.CmdConfig) (string, error) {
    // 判断是否启用动态补全
    if config.DynamicCompletion {

        return generateDynamicScript(cmd, shell)

// generateDynamicScript 生成动态补全脚本
func generateDynamicScript(cmd types.Command, shell string) (string, error) {
```
