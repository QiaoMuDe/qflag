# qflag 动态补全方案 V4

## 设计思路

之前的方案存在一个问题：动态补全需要获取完整的命令树和解析后的参数，但如果在 `--completion` 标志的处理器中执行，解析流程会被中断，无法获取完整的上下文。

新方案的核心思想：
1. **生成脚本**：仍然通过 `--completion` 标志生成补全脚本（静态或动态）
2. **执行补全**：通过内置的子命令 `__complete` 来执行动态补全
3. **子命令内置**：`__complete` 子命令在库内部预先定义好，启用动态补全时自动添加到主命令
4. **完整解析**：调用 `__complete` 时，完整的解析流程会执行，可以获取完整的命令树和解析后的参数

## 架构设计

### 数据流转

```
阶段1: 生成补全脚本
┌─────────────┐     --completion=bash      ┌──────────────────┐
│   用户执行   │ ─────────────────────────→ │  生成补全脚本     │
│   --comp-mode│                            │  (静态或动态)     │
│   =dynamic   │                            └──────────────────┘
└─────────────┘                                       ↓
                                            ┌──────────────────┐
                                            │  脚本包含调用逻辑  │
                                            │  __complete 子命令 │
                                            └──────────────────┘

阶段2: 执行动态补全（用户按 Tab）
┌─────────────┐     调用程序                ┌──────────────────┐
│  Shell 脚本  │ ─────────────────────────→ │  解析 __complete  │
│   (按Tab)    │    myapp __complete ...    │  子命令及参数      │
└─────────────┘                            └──────────────────┘
                                                    ↓
                                            ┌──────────────────┐
                                            │  完整解析流程执行  │
                                            │  获取命令树和参数  │
                                            └──────────────────┘
                                                    ↓
                                            ┌──────────────────┐
                                            │  计算补全结果      │
                                            │  输出到 stdout    │
                                            └──────────────────┘
                                                    ↓
                                            ┌──────────────────┐
                                            │  Shell 解析输出   │
                                            │  展示补全候选      │
                                            └──────────────────┘
```

## 实现方案

### 1. 新增内置子命令 `__complete`

**位置**: `internal/builtin/complete_cmd.go`

```go
package builtin

import (
    "fmt"
    "os"
    "strings"
    
    "gitee.com/MM-Q/qflag/internal/completion"
    "gitee.com/MM-Q/qflag/internal/types"
)

// CompleteCmd 动态补全子命令的名称
const CompleteCmdName = "__complete"

// CreateCompleteCmd 创建动态补全子命令
//
// 返回值:
//   - types.Command: 动态补全子命令实例
//
// 功能说明:
//   - 创建一个隐藏的子命令，用于执行动态补全
//   - 接收当前命令行作为参数
//   - 计算并返回补全结果
func CreateCompleteCmd() types.Command {
    cmd := &completeCmd{
        longName:  CompleteCmdName,
        shortName: "",
        desc:      "内部命令：执行动态补全",
        hidden:    true, // 隐藏子命令，不在帮助中显示
    }
    return cmd
}

// completeCmd 动态补全子命令实现
type completeCmd struct {
    longName  string
    shortName string
    desc      string
    hidden    bool
}

func (c *completeCmd) Name() string           { return c.longName }
func (c *completeCmd) LongName() string       { return c.longName }
func (c *completeCmd) ShortName() string      { return c.shortName }
func (c *completeCmd) Desc() string           { return c.desc }
func (c *completeCmd) IsHidden() bool         { return c.hidden }
func (c *completeCmd) IsRootCmd() bool        { return false }

// Execute 执行动态补全
//
// 参数:
//   - args: 命令行参数（即用户输入的完整命令行）
//
// 功能说明:
//   - 解析 args 获取当前命令上下文
//   - 根据上下文计算补全候选
//   - 输出格式化的补全结果
func (c *completeCmd) Execute(rootCmd types.Command, args []string) {
    // args 格式: [shell类型, 当前输入的词, 前一个词, ...命令行参数]
    if len(args) < 3 {
        fmt.Fprintln(os.Stderr, "用法: __complete <shell> <wordToComplete> <previousWord> [args...]")
        os.Exit(1)
    }
    
    shellType := args[0]
    wordToComplete := args[1]
    previousWord := args[2]
    cmdArgs := args[3:] // 实际的命令行参数
    
    // 构建补全上下文
    ctx := completion.BuildContext(rootCmd, shellType, wordToComplete, previousWord, cmdArgs)
    
    // 执行补全计算
    result := completion.Execute(ctx)
    
    // 输出结果
    completion.PrintResult(result)
    
    os.Exit(0)
}
```

### 2. 修改动态补全脚本模板

**Bash 动态补全模板** (`templates/bash_dynamic.tmpl`):

```bash
# {{.ProgramName}} dynamic completion script
# Generated by qflag

_{{.ProgramName}}_dynamic_completion() {
    local cur prev words cword
    _init_completion || return

    # 调用 __complete 子命令获取补全结果
    # 格式: program __complete <shell> <当前词> <前一个词> <完整命令行>
    local completions
    completions=$({{.ProgramName}} __complete bash "$cur" "$prev" ${words[@]} 2>/dev/null)
    
    # 解析结果（最后一行是 :指令）
    local directive
    directive=$(echo "$completions" | tail -n 1)
    
    # 提取候选值（去掉最后一行指令）
    local candidates
    candidates=$(echo "$completions" | head -n -1)
    
    # 设置补全结果
    COMPREPLY=($(compgen -W "$candidates" -- "$cur"))
    
    return 0
}

# 注册补全函数
complete -F _{{.ProgramName}}_dynamic_completion {{.ProgramName}}
```

**PowerShell 动态补全模板** (`templates/pwsh_dynamic.tmpl`):

```powershell
# {{.ProgramName}} dynamic completion script
# Generated by qflag

Register-ArgumentCompleter -Native -CommandName {{.ProgramName}} -ScriptBlock {
    param($wordToComplete, $commandAst, $cursorPosition)
    
    # 获取完整的命令行参数
    $words = $commandAst.CommandElements | ForEach-Object { $_.ToString() }
    
    # 获取前一个词
    $prev = ""
    if ($words.Count -gt 1) {
        $prev = $words[$words.Count - 2]
    }
    
    # 调用 __complete 子命令获取补全结果
    # 格式: program __complete <shell> <当前词> <前一个词> <完整命令行>
    $completions = & {{.ProgramName}} __complete pwsh $wordToComplete $prev @words 2>$null
    
    if (-not $completions) {
        return
    }
    
    # 解析结果（最后一行是 :指令）
    $lines = $completions -split "`n"
    $directive = $lines[-1]
    $candidates = $lines[0..($lines.Count - 2)]
    
    # 返回候选值
    foreach ($candidate in $candidates) {
        if ($candidate -match "^$wordToComplete") {
            # 处理带描述的格式: value`tdescription
            $parts = $candidate -split "`t", 2
            [System.Management.Automation.CompletionResult]::new(
                $parts[0],
                $parts[0],
                'ParameterValue',
                if ($parts[1]) { $parts[1] } else { $parts[0] }
            )
        }
    }
}
```

### 3. 修改 `completion` 包

**位置**: `internal/completion/context.go` (新增)

```go
package completion

import "gitee.com/MM-Q/qflag/internal/types"

// Context 补全上下文
type Context struct {
    RootCmd        types.Command  // 根命令
    CurrentCmd     types.Command  // 当前命令（解析后的）
    ShellType      string         // Shell 类型
    WordToComplete string         // 当前正在补全的词
    PreviousWord   string         // 前一个词
    Args           []string       // 命令行参数
    Flags          []types.Flag   // 当前命令的标志列表
    SubCmds        []types.Command // 当前命令的子命令列表
}

// BuildContext 构建补全上下文
//
// 参数:
//   - rootCmd: 根命令
//   - shellType: Shell 类型
//   - wordToComplete: 当前词
//   - previousWord: 前一个词
//   - args: 命令行参数
//
// 返回值:
//   - *Context: 补全上下文
func BuildContext(rootCmd types.Command, shellType, wordToComplete, previousWord string, args []string) *Context {
    ctx := &Context{
        RootCmd:        rootCmd,
        ShellType:      shellType,
        WordToComplete: wordToComplete,
        PreviousWord:   previousWord,
        Args:           args,
    }
    
    // 根据 args 解析出当前命令
    ctx.CurrentCmd = resolveCurrentCmd(rootCmd, args)
    ctx.Flags = ctx.CurrentCmd.Flags()
    ctx.SubCmds = ctx.CurrentCmd.SubCmds()
    
    return ctx
}

// resolveCurrentCmd 根据参数解析当前命令
func resolveCurrentCmd(rootCmd types.Command, args []string) types.Command {
    current := rootCmd
    for _, arg := range args {
        // 遇到标志就停止
        if strings.HasPrefix(arg, "-") {
            break
        }
        // 检查是否是子命令
        if subCmd, ok := current.GetSubCmd(arg); ok {
            current = subCmd
        } else {
            break
        }
    }
    return current
}
```

**位置**: `internal/completion/execute.go` (新增)

```go
package completion

import (
    "fmt"
    "strings"
    
    "gitee.com/MM-Q/qflag/internal/types"
)

// Execute 执行补全计算
//
// 参数:
//   - ctx: 补全上下文
//
// 返回值:
//   - types.CompletionResult: 补全结果
func Execute(ctx *Context) types.CompletionResult {
    // 1. 检查是否有自定义补全函数
    if ctx.PreviousWord != "" {
        // 去掉前缀的 -
        flagName := strings.TrimLeft(ctx.PreviousWord, "-")
        if completer, ok := ctx.CurrentCmd.GetCompleter(flagName); ok {
            return completer(buildCompleterContext(ctx))
        }
    }
    
    // 2. 使用默认补全逻辑
    return defaultComplete(ctx)
}

// defaultComplete 默认补全逻辑
func defaultComplete(ctx *Context) types.CompletionResult {
    var completions []types.Completion
    
    // 根据上下文决定补全什么
    if isCompletingFlagValue(ctx) {
        // 补全标志值
        completions = completeFlagValue(ctx)
    } else if strings.HasPrefix(ctx.WordToComplete, "-") {
        // 补全标志名
        completions = completeFlagName(ctx)
    } else {
        // 补全子命令
        completions = completeSubCommand(ctx)
    }
    
    return types.CompletionResult{
        Completions: completions,
        Directive:   0,
    }
}

// isCompletingFlagValue 判断是否正在补全标志值
func isCompletingFlagValue(ctx *Context) bool {
    return ctx.PreviousWord != "" && strings.HasPrefix(ctx.PreviousWord, "-")
}

// completeFlagValue 补全标志值
func completeFlagValue(ctx *Context) types.CompletionResult {
    // 实现标志值补全逻辑
    // 1. 获取标志类型
    // 2. 如果是枚举类型，返回枚举值
    // 3. 其他类型返回空（由 Shell 处理文件补全）
    return types.CompletionResult{}
}

// completeFlagName 补全标志名
func completeFlagName(ctx *Context) []types.Completion {
    var completions []types.Completion
    for _, flag := range ctx.Flags {
        name := flag.LongName()
        if strings.HasPrefix(name, ctx.WordToComplete) {
            completions = append(completions, types.Completion{
                Value:       name,
                Description: flag.Desc(),
            })
        }
    }
    return completions
}

// completeSubCommand 补全子命令
func completeSubCommand(ctx *Context) []types.Completion {
    var completions []types.Completion
    for _, subCmd := range ctx.SubCmds {
        name := subCmd.Name()
        if strings.HasPrefix(name, ctx.WordToComplete) {
            completions = append(completions, types.Completion{
                Value:       name,
                Description: subCmd.Desc(),
            })
        }
    }
    return completions
}

// PrintResult 打印补全结果
func PrintResult(result types.CompletionResult) {
    for _, comp := range result.Completions {
        if comp.Description != "" {
            fmt.Printf("%s\t%s\n", comp.Value, comp.Description)
        } else {
            fmt.Println(comp.Value)
        }
    }
    fmt.Printf(":%d\n", result.Directive)
}
```

```go
func (m *BuiltinFlagManager) RegisterBuiltinFlags(cmd types.Command) error {
```

### 5. 修改 `cmd` 包

**位置**: `internal/cmd/cmd.go`

```go
// SetDynamicCompletion 设置是否启用动态补全
//
// 参数:
//   - enable: 是否启用
//
// 功能说明:
//   - 启用后，会注册 __complete 子命令
//   - 动态补全脚本会调用此子命令
func (c *Cmd) SetDynamicCompletion(enable bool) {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.config.DynamicCompletion = enable
}

// RegisterCompleter 注册自定义补全函数
//
// 参数:
//   - flagName: 标志名称
//   - completer: 补全函数
func (c *Cmd) RegisterCompleter(flagName string, completer types.DynamicCompleter) {
    c.mu.Lock()
    defer c.mu.Unlock()
    if c.completers == nil {
        c.completers = make(map[string]types.DynamicCompleter)
    }
    c.completers[flagName] = completer
}

// GetCompleter 获取自定义补全函数
func (c *Cmd) GetCompleter(flagName string) (types.DynamicCompleter, bool) {
    c.mu.RLock()
    defer c.mu.RUnlock()
    completer, ok := c.completers[flagName]
    return completer, ok
}
```

## 使用方式

### 用户代码示例

```go
package main

import (
    "os"
    "gitee.com/MM-Q/qflag"
    "gitee.com/MM-Q/qflag/internal/types"
)

func main() {
    // 创建命令
    root := qflag.NewCmd("myapp", "我的应用", qflag.ExitOnError)
    
    // 【关键】启用动态补全
    // 这会注册 __complete 子命令
    root.SetDynamicCompletion(true)
    
    // 启用补全标志（用于生成补全脚本）
    root.SetCompletion(true)
    
    // 创建子命令
    deploy := qflag.NewCmd("deploy", "部署应用", qflag.ExitOnError)
    deploy.String("env", "e", "部署环境", "dev")
    
    // 注册自定义补全函数
    deploy.RegisterCompleter("env", func(ctx types.CompletionContext) types.CompletionResult {
        return types.CompletionResult{
            Completions: []types.Completion{
                {Value: "dev", Description: "开发环境"},
                {Value: "prod", Description: "生产环境"},
            },
        }
    })
    
    root.AddSubCmds(deploy)
    
    // 执行
    if err := root.ParseAndRoute(os.Args[1:]); err != nil {
        os.Exit(1)
    }
}
```

### 补全执行流程

```
1. 用户在 Shell 中输入: myapp deploy --env <Tab>

2. Shell 补全脚本执行:
   myapp __complete bash "" "--env" "myapp" "deploy" "--env" ""
   
   参数说明:
   - bash: Shell 类型
   - "": 当前词（空表示刚输入空格）
   - "--env": 前一个词
   - "myapp" "deploy" "--env" "": 完整命令行

3. qflag 解析 __complete 子命令:
   - 识别为内置子命令
   - 执行补全逻辑
   - 返回补全结果

4. 输出格式:
   dev	开发环境
   prod	生产环境
   :0

5. Shell 解析输出并展示给用户
```

## 优势

1. **完整解析流程**: `__complete` 作为普通子命令执行，可以享受完整的解析流程
2. **获取完整上下文**: 可以获取解析后的命令树、参数值等
3. **支持自定义补全**: 用户可以注册自定义补全函数
4. **向后兼容**: 不影响现有的静态补全功能
5. **隐藏子命令**: `__complete` 是隐藏子命令，不会在帮助中显示

## 注意事项

1. `__complete` 是保留子命令名，用户不应创建同名子命令
2. 动态补全会启动程序，对于大型程序可能有性能影响
3. 自定义补全函数应该快速执行，避免阻塞补全流程
